// Slice/view zero-copy
    AArray<T> view(
        const std::vector<size_t>& start,
        const std::vector<size_t>& stop,
        const std::vector<size_t>& step) const
    {
        if(start.size() != p_shape.rank() || stop.size() != p_shape.rank() || step.size() != p_shape.rank())
            throw std::invalid_argument("view: dimension mismatch");

        // Calcola dimensioni della vista
        std::vector<size_t> viewDims(p_shape.rank());
        for(size_t i = 0; i < p_shape.rank(); ++i){
            if(step[i] == 0) throw std::invalid_argument("view: step cannot be zero");
            if(stop[i] <= start[i]) throw std::invalid_argument("view: stop must be > start");
            viewDims[i] = (stop[i] - start[i] + step[i] - 1) / step[i];
        }

        // Costruisci la shape della vista
        AShape newShape(viewDims);

        // Strides della vista
        std::vector<size_t> newStrides(p_shape.rank());
        for(size_t i = 0; i < p_shape.rank(); ++i){
            newStrides[i] = p_shape.strides()[i] * step[i];
        }

        // Offset lineare iniziale
        size_t linearOffset = 0;
        const auto oldStrides = p_shape.strides();
        for(size_t i = 0; i < p_shape.rank(); ++i)
            linearOffset += start[i] * oldStrides[i];

        // Crea nuovo AArray come view
        AArray<T> result;
        result.p_shape = newShape;
        result.p_dataPtr = p_dataPtr;      // condivisione dei dati
        result.p_offset = linearOffset;    // nuovo offset
        result.p_strides = newStrides;     // nuovi strides calcolati
        result.p_isView = true;

        return result;
    }
