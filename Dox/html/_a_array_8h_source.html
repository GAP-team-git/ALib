122 AArray<T> slice(const std::vector<size_t>& start,
123 const std::vector<size_t>& stop,
124 const std::vector<size_t>& step={}) const {
125 std::vector<size_t> sstep = step.empty()? std::vector<size_t>(p_shape.rank(),1):step;
126 AShape newShape = p_shape.sliceShape(start,stop,sstep);
127 AArray<T> result(newShape);

147 AArray<T> transpose(const std::vector<size_t>& axes={}) const {
148 AShape tshape = p_shape;
149 if(axes.empty()){
150 std::vector<size_t> perm(p_shape.rank());
151 for(size_t i=0;i<p_shape.rank();++i) perm[i]=p_shape.rank()-1-i;
152 tshape.transpose(perm);
153 } else tshape.transpose(axes);
154 AArray<T> result;
155 result.p_shape = tshape;
156 result.p_dataPtr = p_dataPtr; // shared view
157 return result;

201 // -------------------
202 // Binary ops with broadcasting
203 // -------------------
204 template<typename Op>
205 static AArray<T> broadcastBinaryOp(const AArray<T>& lhs,const AArray<T>& rhs, Op op){
206 AShape resultShape = AShape::broadcast(lhs.shape(),rhs.shape());
207 AArray<T> result(resultShape);
208
209 const auto& lhsDims=lhs.shape().dims();
210 const auto& rhsDims=rhs.shape().dims();
211 const auto& lhsStrides=lhs.shape().strides();
212 const auto& rhsStrides=rhs.shape().strides();
213 const auto& resDims=resultShape.dims();
214
215 auto& resData = result.p_dataPtr;
216 const auto& lhsData = lhs.p_dataPtr;
217 const auto& rhsData = rhs.p_dataPtr;
218
219 size_t rankRes = resultShape.rank();
220 size_t rankL = lhs.shape().rank();
221 size_t rankR = rhs.shape().rank();
222
223 for(size_t i=0;i<resultShape.totalSize();++i){
224 size_t idx=i;
225 size_t lhsIdx=0, rhsIdx=0;
226 for(int d=int(rankRes)-1;d>=0;--d){
227 size_t cur = idx % resDims[d];
228
229 size_t ldim = (d>=rankRes-rankL)?lhsDims[d-(rankRes-rankL)]:1;
230 size_t rdim = (d>=rankRes-rankR)?rhsDims[d-(rankRes-rankR)]:1;
231
232 size_t lstride=(d>=rankRes-rankL)?lhsStrides[d-(rankRes-rankL)]:0;
233 size_t rstride=(d>=rankRes-rankR)?rhsStrides[d-(rankRes-rankR)]:0;
234
235 lhsIdx += (ldim==1?0:cur*lstride);
236 rhsIdx += (rdim==1?0:cur*rstride);
237
238 idx/=resDims[d];
239 }
240 (*resData)[i] = op((*lhsData)[lhsIdx], (*rhsData)[rhsIdx]);
241 }
242
243 return result;
244 }
