38 if(idx.size() != rank())
39 throw std::runtime_error("AShape::flatIndex rank mismatch");

51 // Slice support

53 AShape sliceShape(const std::vector<size_t>& start,
54 const std::vector<size_t>& stop,
55 const std::vector<size_t>& step) const {
56 if(start.size()!=rank() || stop.size()!=rank() || step.size()!=rank())
57 throw std::invalid_argument("sliceShape: size mismatch");
58 std::vector<size_t> newDims(rank());
59 for(size_t i=0;i<rank();++i){
60 if(step[i]==0) throw std::invalid_argument("sliceShape: step=0");
61 newDims[i] = (stop[i]<=start[i]?0:(stop[i]-start[i]+step[i]-1)/step[i]);
62 }
63 return AShape(newDims);
64 }

66 size_t flatIndexFromSlice(size_t idx, const std::vector<size_t>& start, const std::vector<size_t>& step) const {
67 size_t linear = 0;
68 for(int d=int(rank())-1; d>=0; --d){
69 size_t cur = idx % dims()[d];
70 linear += (start[d] + cur*step[d])*strides()[d];
71 idx /= dims()[d];
72 }
73 return linear;
74 }

79 void transpose(const std::vector<size_t>& axes) {
80 if(axes.size()!=rank()) throw std::invalid_argument("transpose: axes mismatch");
81 std::vector<size_t> newDims(rank());
82 for(size_t i=0;i<rank();++i) newDims[i]=p_dims[axes[i]];
83 p_dims=newDims;

87 // -------------------
88 // Broadcasting
89 // -------------------
90 static AShape broadcast(const AShape& a, const AShape& b){
91 size_t r1=a.rank(), r2=b.rank();
92 size_t rmax=std::max(r1,r2);
93 std::vector<size_t> dims(rmax,1);
94 for(size_t i=0;i<rmax;++i){
95 size_t da=(i<rmax-r1?1:a.p_dims[i-(rmax-r1)]);
96 size_t db=(i<rmax-r2?1:b.p_dims[i-(rmax-r2)]);
97 if(da!=db && da!=1 && db!=1) throw std::invalid_argument("broadcast: incompatible shapes");
98 dims[i]=std::max(da,db);
99 }
100 return AShape(dims);
101 }

110 void calcStrides(){
111 p_strides.resize(rank());

113 p_strides[rank()-1]=1;

115 p_strides[i] = p_strides[i+1]*p_dims[i+1];
116 }
