94 static std::string generateOID() {
95 static thread_local std::mt19937_64 rng(
96 std::chrono::high_resolution_clock::now().time_since_epoch().count()
97 );
98 std::uniform_int_distribution<uint64_t> dist(0, 9999999999ULL);
99 std::ostringstream oss;
100 for(int i = 0; i < 3; ++i) {
101 oss << std::setw(10) << std::setfill('0') << dist(rng);
102 if(i != 2) oss << ".";
103 }
104 return oss.str();
105 }

142 virtual std::istream& m_parse(std::istream& is) {
143 std::string oid;
144 bool valid = false;
145 int errInt = 0;
146 uint32_t version = 0;
147 is >> version >> oid >> valid >> errInt;
148 if (is.good()) {
149 p_version = version;
150 p_oid = std::move(oid);
151 p_valid = valid;
152 p_error = static_cast<AErrors>(errInt);
153 }
154 return is;

158 [[nodiscard("AObject m_write")]] virtual bool m_write(std::ostream& os) const {
159 os.write(reinterpret_cast<const char*>(&p_magic), sizeof(p_magic));
160 os.write(reinterpret_cast<const char*>(&p_version), sizeof(p_version));
161
162 uint32_t oidLen = static_cast<uint32_t>(p_oid.size());
163 os.write(reinterpret_cast<const char*>(&oidLen), sizeof(oidLen));
164 if (oidLen > 0) os.write(p_oid.data(), oidLen);
165
166 uint8_t validByte = p_valid ? 1u : 0u;
167 os.write(reinterpret_cast<const char*>(&validByte), sizeof(validByte));
168
169 int32_t err = static_cast<int32_t>(p_error);
170 os.write(reinterpret_cast<const char*>(&err), sizeof(err));
171
172 return os.good();
173 }

175 [[nodiscard("AObject m_read")]] virtual bool m_read(std::istream& is) {
176 uint32_t magic = 0;
177 is.read(reinterpret_cast<char*>(&magic), sizeof(magic));
178 if (magic != AOBJ_MAGIC) return false;
179
180 uint32_t version = 0;
181 is.read(reinterpret_cast<char*>(&version), sizeof(version));
182 if (!is.good()) return false;
183 p_version = version;
184
185 uint32_t oidLen = 0;
186 is.read(reinterpret_cast<char*>(&oidLen), sizeof(oidLen));
187 if (!is.good()) return false;
188
189 std::string oid; oid.resize(oidLen);
190 if (oidLen > 0) {
191 is.read(oid.data(), oidLen);
192 if (!is.good()) return false;
193 }
194
195 uint8_t validByte = 0;
196 is.read(reinterpret_cast<char*>(&validByte), sizeof(validByte));
197 if (!is.good()) return false;
198
199 int32_t err = 0;
200 is.read(reinterpret_cast<char*>(&err), sizeof(err));
201 if (!is.good()) return false;
202
203 p_oid = std::move(oid);
204 p_valid = (validByte != 0);
205 p_error = static_cast<AErrors>(err);
206
207 return true;
208 }
